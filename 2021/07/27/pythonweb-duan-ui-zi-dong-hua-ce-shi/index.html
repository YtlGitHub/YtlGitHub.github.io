import unittest
from unittestreport import TestRunner
from common.configTime import *
from product_info.info import *


class AllTest():
    def __init__(self):
        pass

    def run(self):
        """иҝҗиЎҢи„ҡжң¬з”ЁдҫӢз”ҹжҲҗжөӢиҜ•жҠҘе‘Ҡ"""
        a = config_report_time_path()  # й…ҚзҪ®еҢ…еҗҚж—¶й—ҙе’Ңи·Ҝеҫ„
        case_push = a[0]  # з”ЁдҫӢи·Ҝеҫ„
        report_name = a[1]  # жҠҘе‘ҠеҗҚ
        report_push = a[2]  # жҠҘе‘Ҡи·Ҝеҫ„
        suite = unittest.defaultTestLoader.discover(case_push)  # еҠ иҪҪжөӢиҜ•з”ЁдҫӢ
        TestRunner(suite,
                   filename=report_name,
                   report_dir=report_push,
                   title="жөӢиҜ•жҠҘе‘Ҡ",
                   tester="дҪ иҮӘе·ұзҡ„еҗҚеӯ—",
                   desc=f"жҠҘе‘ҠдҝЎжҒҜ",
                   templates=1).rerun_run(count=0, interval=2)  # еӨұиҙҘйҮҚиҝҗиЎҢж¬Ўж•°


if __name__ == '__main__':
    run = AllTest()
    run.run()
